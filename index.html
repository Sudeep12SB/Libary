from flask import Flask, request, jsonify import sqlite3 from datetime import datetime app = Flask(__name__) # Connect to SQLite database (library.db) def get_db_connection(): conn = sqlite3.connect('library.db') conn.row_factory = sqlite3.Row return conn # Register user route @app.route('/register', methods=['POST']) def register(): name = request.form.get('name') branch = request.form.get('branch') reg_no = request.form.get('reg_no') # Insert user into the database conn = get_db_connection() conn.execute('INSERT INTO users (name, branch, reg_no) VALUES (?, ?, ?)', (name, branch, reg_no)) conn.commit() conn.close() return jsonify({'message': f'User {name} registered successfully!'}) # Fetch available books route @app.route('/available_books', methods=['GET']) def available_books(): conn = get_db_connection() books = conn.execute('SELECT title FROM books WHERE is_available = 1').fetchall() conn.close() available_books = [book['title'] for book in books] return jsonify({'available_books': available_books}) # Add a book route @app.route('/add_book', methods=['POST']) def add_book(): title = request.form.get('title') # Insert new book into the database conn = get_db_connection() conn.execute('INSERT INTO books (title) VALUES (?)', (title,)) conn.commit() conn.close() return jsonify({'message': f'Book "{title}" added successfully!'}) # Borrow a book route @app.route('/borrow_book', methods=['POST']) def borrow_book(): book_title = request.form.get('book_title') user_name = request.form.get('user_id') # User identifier as the name conn = get_db_connection() # Get the book ID if available book = conn.execute('SELECT id FROM books WHERE title = ? AND is_available = 1', (book_title,)).fetchone() if not book: conn.close() return jsonify({'message': f'Book "{book_title}" is not available!'}) # Get the user ID user = conn.execute('SELECT id FROM users WHERE name = ?', (user_name,)).fetchone() if not user: conn.close() return jsonify({'message': 'User not found!'}) # Mark the book as borrowed and update borrowed_books table conn.execute('UPDATE books SET is_available = 0 WHERE id = ?', (book['id'],)) conn.execute('INSERT INTO borrowed_books (user_id, book_id, borrow_date) VALUES (?, ?, ?)', (user['id'], book['id'], datetime.now())) conn.commit() conn.close() return jsonify({'message': f'Book "{book_title}" borrowed successfully!'}) if __name__ == '__main__': app.run(debug=True)
